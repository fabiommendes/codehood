    def check_errors(self) -> Iterable[Issue]:
        raise NotImplementedError

    def validate(self, severity=Severity.STRICT):
        """
        Raise a ValidationError if any errors were found.
        """
        errors = []
        for error in self.check_errors():
            if error >= severity:
                errors.append(error)
        if errors:
            raise ValidationError(errors)

class ValidationError (errors: Iterable[Issue])
Expand source code 
class ValidationError(ValueError):
    @property
    def errors(self):
        return self.args[0]

    def __init__(self, errors: Iterable[Issue]):
        super().__init__([*errors])

    def __str__(self):
        match self.errors:
            case []:
                return "no error found!"
            case [error]:
                return str(error)
            case errors:
                return "\n".join("    " + str(error) for error in errors)
